# Что такое requirements.txt?
# Это стандартный файл Python для описания зависимостей.
# pip install -r requirements.txt установит все перечисленные пакеты.
#
# Формат:
#   имя_пакета==версия    # Точная версия
#   имя_пакета>=версия    # Минимальная версия
#   имя_пакета            # Любая версия (не рекомендуется)
# ═══════════════════════════════════════════════════════════════════════════
